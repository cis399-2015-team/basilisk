### Samba server and client setup instructions

## Server:

# Install the software
sudo apt-get install samba

# Setup a user
sudo smbpasswd -a <user_name>

# Create the share directory
mkdir /home/<user_name>/<folder_name>

# Create a copy of the conf file
sudo cp /etc/samba/smb.conf ~

# Edit the conf to to add the shared directory, add this to the end
[<folder_name>]
path = /home/<user_name>/<folder_name>
available = yes
valid users = <user_name>
read only = no
browseable = yes
public = yes
writable = yes

# Restart samba
sudo service smbd restart

## Client:

# Install the software
sudo apt-get install cifs-utils

# Create samba group and add a user
sudo groupadd samba
sudo adduser <user_name> samba

# Edit sudoers to add the new group so its members can mount SMB shares
sudo visudo

# Add this line in the "group" section under %admin
%samba   ALL=(ALL) /bin/mount,/bin/umount,/sbin/mount.cifs,/sbin/umount.cifs

# Create the mount location
mkdir ~/mnt

# Mount the shared folder
sudo mount -t cifs //<ip_address>/<share> ~/mnt -o username=<user_name>,noexec

# Unmount the shared folder
sudo umount ~/mnt

# More info
https://help.ubuntu.com/community/Samba/SambaClientGuide
